# Redirect

```js
import React from "react";
import Helmet from "react-helmet";

const Redirect = props => {
  const {
    url
    // pageContext: { url }
  } = props;
  return (
    <Helmet>
      <meta http-equiv="refresh" content={`0;url=${url}`} />
    </Helmet>
  );
};

export default Redirect;
```
